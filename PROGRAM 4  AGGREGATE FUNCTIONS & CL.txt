 SELECT MIN(SAL),MAX(SAL),AVG(SAL),SUM(SAL) FROM EMP;

SELECT MAX(SAL) FROM EMP WHERE JOB='CLERK'; 

 SELECT SUM(SAL) FROM EMP WHERE DEPTNO=20 AND JOB='ANALYST'; 


 SELECT DEPTNO,COUNT(*) FROM EMP   GROUP BY DEPTNO;

 SELECT JOB,COUNT(*) FROM EMP GROUP BY JOB;

SELECT JOB,MIN(SAL), FROM EMP GROUP BY JOB;


 SELECT JOB,MIN(SAL),MAX(SAL),AVG(SAL),SUM(SAL) FROM EMP GROUP BY JOB; 


 SELECT DEPTNO,JOB,COUNT(*) FROM EMP  GROUP BY DEPTNO,JOB;

 SELECT DEPTNO,JOB,COUNT(*) FROM EMP WHERE DEPTNO=20 GROUP BY DEPTNO,JOB; 


 SELECT DEPTNO,JOB,COUNT(*) FROM EMP  
      WHERE DEPTNO=20  
      GROUP BY DEPTNO,JOB 
      HAVING COUNT(*)>1 
     ORDER BY JOB;
